■otemotoディレクトリ作成
$ cd /
$ sudo mkdir blueprint
$ cd /blueprint
$ sudo mkdir docker
$ cd /blueprint/docker/

■nginxのdockerfile作成
$ cd /blueprint/docker/
$ sudo mkdir nginx
$ cd /blueprint/docker/nginx/
$ sudo touch Dockerfile-nginx
$ sudo vim Dockerfile-nginx
FROM nginx:1.21
COPY ./default.conf /etc/nginx/conf.d/default.conf
$ sudo touch default.conf
$ sudo vim default.conf
server {
    listen       80;
    server_name  localhost;
    location / {
        root /usr/share/nginx/html;
        index index.html;
    }
}

■docker-compose.yml作成
$ cd /blueprint
$ sudo touch docker-compose.yml
$ sudo vim docker-compose.yml
version: '3'
services:
  web:
    build:
      context: ./docker/nginx
      dockerfile: Dockerfile-nginx
    image: blueprint-nginx
    container_name: blueprint-nginx
    ports:
      - 1080:80

■publicフォルダマウント
$ cd /blueprint
$ sudo mkdir public
$ sudo chmod 777 /blueprint/public

■コンテナ起動
$ cd /brueprint
$ docker-compose up --build
$ sudo vim docker-compose.yml
version: '3'
services:
  web:
    build:
      context: ./docker/nginx
      dockerfile: Dockerfile-nginx
    image: blueprint-nginx
    container_name: blueprint-nginx
    ports:
      - 1080:80
    volumes:
      - ./public:/usr/share/nginx/html
