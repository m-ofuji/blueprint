■nginxのdockerfile作成
$ cd ./services
$ sudo vi
version: "3"
services:
  proxy:
    image: jwilder/nginx-proxy
    container_name: proxy
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - /var/run/docker.sock:/tmp/docker.sock:ro
    restart: always
    logging:
      options:
        max-size: 5m
networks:
  default:
    external:
      name: services

■app側nginxの設定変更
$ sudo vim docker-compose.yml
version: '3'
services:
  web:
    build:
      context: ./docker/nginx
      dockerfile: Dockerfile-nginx
    image: blueprint-nginx
    environment:
      VIRTAL_HOST: toporis.net
    container_name: blueprint-nginx
    volumes:
      - ./public:/usr/share/nginx/html
networks:
  default:
    external:
      name: services


■ネットワーク作成
$ cd /services
$ docker network create --driver bridge services

■リバースプロキシ起動
$ cd /services
$ docker-compose up -d
